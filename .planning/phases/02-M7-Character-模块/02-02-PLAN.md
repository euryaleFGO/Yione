    ```python
    from motor.motor_asyncio import AsyncIOMotorDatabase
    from app.schemas.chat import SessionInfo

    class MongoSessionRepo:
        def __init__(self, db: AsyncIOMotorDatabase | None):
            self._db = db
            self._fallback_store: dict[str, SessionInfo] = {}

        async def create(self, session: SessionInfo) -> SessionInfo:
            if self._db is None:
                # Dev fallback: 内存模式
                self._fallback_store[session.session_id] = session
                return session
            doc = session.model_dump(mode="json")
            await self._db.sessions.insert_one(doc)
            return session

        async def get(self, session_id: str) -> SessionInfo | None:
            if self._db is None:
                return self._fallback_store.get(session_id)
            doc = await self._db.sessions.find_one({"session_id": session_id})
            if doc is None:
                return None
            doc.pop("_id", None)
            return SessionInfo.model_validate(doc)
    ```

    ```python
    from motor.motor_asyncio import AsyncIOMotorDatabase
    from app.services.tenant_service import Tenant

    class MongoTenantRepo:
        def __init__(self, db: AsyncIOMotorDatabase | None):
            self._db = db
            self._dev_fallback = db is None

        async def verify_api_key(self, api_key: str) -> Tenant | None:
            if not api_key:
                return None
            if self._dev_fallback:
                # Dev fallback: 任意 key 通过
                from app.services.tenant_service import _hash_key
                return Tenant(
                    id="demo",
                    api_key_hash=_hash_key(api_key),
                    allowed_origins=[],
                    character_whitelist=[],
                    daily_quota=2000,
                    scope=["chat", "embed"],
                )
            import hashlib
            digest = hashlib.sha256(api_key.encode("utf-8")).hexdigest()
            doc = await self._db.tenants.find_one({"api_key_hash": digest})
            if doc is None:
                return None
            doc.pop("_id", None)
            return Tenant(**doc)
    ```

    ```python
    from motor.motor_asyncio import AsyncIOMotorDatabase
    from app.domain.speaker import Speaker
    from app.schemas.speaker import SpeakerProfile
    from app.repositories.speaker_repo import SpeakerRepository

    class MongoSpeakerRepo:
        def __init__(self, db: AsyncIOMotorDatabase | None):
            self._db = db
            # 如果无 MongoDB，降级到 JSON 文件
            if db is None:
                from app.repositories.speaker_repo import JsonFileSpeakerRepo
                self._fallback = JsonFileSpeakerRepo()
            else:
                self._fallback = None

        async def list_all(self) -> list[Speaker]:
            if self._fallback:
                return self._fallback.list_all()
            cursor = self._db.speakers.find()
            docs = await cursor.to_list(length=1000)
            return [_doc_to_speaker(d) for d in docs]

        async def get(self, speaker_id: str) -> Speaker | None:
            if self._fallback:
                return self._fallback.get(speaker_id)
            doc = await self._db.speakers.find_one({"id": speaker_id})
            if doc is None:
                return None
            return _doc_to_speaker(doc)

        async def save(self, speaker: Speaker) -> None:
            if self._fallback:
                self._fallback.save(speaker)
                return
            doc = _speaker_to_dict(speaker)
            await self._db.speakers.update_one(
                {"id": speaker.id},
                {"$set": doc},
                upsert=True,
            )

        async def delete(self, speaker_id: str) -> bool:
            if self._fallback:
                return self._fallback.delete(speaker_id)
            result = await self._db.speakers.delete_one({"id": speaker_id})
            return result.deleted_count > 0
    ```
